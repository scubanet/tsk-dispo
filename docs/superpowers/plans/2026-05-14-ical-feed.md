# iCal-Feed pro Instructor — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Jeder ATOLL-Instructor abonniert einen persönlichen iCal-Feed-URL in Apple Calendar / Google / Outlook und sieht ab da seine `course_assignments` automatisch im normalen Kalender (one-way, Token-basiert, rotierbar).

**Architecture:** Drei lose gekoppelte Bausteine: (1) Schema-Migration mit `instructors.calendar_token` + Rotate-RPC, (2) Supabase Edge Function `ical-feed` die per Service-Role-Key auf Token auflöst, Assignments lädt und RFC-5545-konforme `.ics` zurückgibt, (3) Frontend-Erweiterung im `MyProfileScreen` mit URL-Anzeige + Copy + Rotate. Frontend ↔ Edge Function: nur via die ausgehende URL, keine direkte Kopplung.

**Tech Stack:** Postgres (Supabase) + pgcrypto · Deno-Edge-Function + std/http + supabase-js · React 18 + TypeScript + i18next · Vitest installiert aber keine Tests in `apps/web/src` (Verifikation über `npm run typecheck` + curl-Tests + manuelle Apple-Calendar-Subscription).

```sql
-- 0095: instructors.calendar_token für iCal-Feed-Subscription
--
-- Pro Instructor ein 24-Byte-Random-Token (32 Zeichen Base64), der als URL-Param
-- für die Edge Function ical-feed dient. Rotierbar via RPC rotate_calendar_token
-- (siehe 0096).

ALTER TABLE instructors ADD COLUMN IF NOT EXISTS calendar_token TEXT;

-- Backfill für bestehende Instructors (idempotent: nur wo noch NULL)
UPDATE instructors
SET calendar_token = encode(gen_random_bytes(24), 'base64')
WHERE calendar_token IS NULL;

-- NOT NULL + UNIQUE erst nach Backfill
ALTER TABLE instructors ALTER COLUMN calendar_token SET NOT NULL;

-- UNIQUE-Constraint nur anlegen wenn nicht schon vorhanden
DO $$ BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint
    WHERE conname = 'instructors_calendar_token_key'
  ) THEN
    ALTER TABLE instructors ADD CONSTRAINT instructors_calendar_token_key UNIQUE (calendar_token);
  END IF;
END $$;

-- Index für Lookup in der Edge Function
CREATE INDEX IF NOT EXISTS idx_instructors_calendar_token
  ON instructors(calendar_token);

COMMENT ON COLUMN instructors.calendar_token IS
  'Random token (24 bytes Base64) für iCal-Feed-Subscription. Rotierbar via RPC.';
```

```sql
-- 0096: RPC rotate_calendar_token() — Self-Service-Token-Rotation
--
-- Frontend ruft diese Function auf wenn der User in MyProfileScreen
-- "Token zurücksetzen" klickt. Function liest auth.uid() und updated nur
-- den eigenen Instructor-Record. Gibt den neuen Token zurück.

CREATE OR REPLACE FUNCTION rotate_calendar_token()
RETURNS TEXT
LANGUAGE plpgsql SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_new_token TEXT;
  v_instructor_id UUID;
BEGIN
  SELECT id INTO v_instructor_id
  FROM instructors
  WHERE auth_user_id = auth.uid();

  IF v_instructor_id IS NULL THEN
    RAISE EXCEPTION 'No instructor record for current user';
  END IF;

  v_new_token := encode(gen_random_bytes(24), 'base64');

  UPDATE instructors
     SET calendar_token = v_new_token
   WHERE id = v_instructor_id;

  RETURN v_new_token;
END;
$$;

GRANT EXECUTE ON FUNCTION rotate_calendar_token() TO authenticated;

COMMENT ON FUNCTION rotate_calendar_token() IS
  'Erzeugt einen neuen calendar_token für den eingeloggten Instructor und gibt ihn zurück. Alte iCal-Subscriptions werden damit ungültig.';
```

```ts
// RFC 5545 helpers für die ical-feed Edge Function.
// Pure functions, keine Supabase-Abhängigkeit.

export interface CalendarEvent {
  /** Stable, deterministic UID — z.B. assignment-<id>-<YYYYMMDD>@dispo.course-director.ch */
  uid: string
  /** Event title */
  summary: string
  /** Optional location (course.location) */
  location?: string | null
  /** Optional multi-line description */
  description?: string | null
  /** ISO date YYYY-MM-DD */
  date: string
  /** Optional zeit-fenster, beide oder keine */
  timeFrom?: string | null  // 'HH:MM:SS'
  timeTo?: string | null    // 'HH:MM:SS'
  /** course.status → CONFIRMED | TENTATIVE */
  status: 'CONFIRMED' | 'TENTATIVE'
  /** Letzte Änderung — ISO timestamp */
  lastModified: string
}

/** Escape RFC 5545 text values (backslash, comma, semicolon, newline) */
export function escapeText(value: string): string {
  return value
    .replace(/\\/g, '\\\\')
    .replace(/,/g, '\\,')
    .replace(/;/g, '\\;')
    .replace(/\r?\n/g, '\\n')
}

/** Fold lines longer than 75 octets per RFC 5545 — CRLF + space continuation. */
export function foldLine(line: string): string {
  if (new TextEncoder().encode(line).length <= 75) return line
  const chunks: string[] = []
  let buf = ''
  let bufLen = 0
  for (const ch of line) {
    const chLen = new TextEncoder().encode(ch).length
    if (bufLen + chLen > 75) {
      chunks.push(buf)
      buf = ' ' + ch
      bufLen = 1 + chLen
    } else {
      buf += ch
      bufLen += chLen
    }
  }
  if (buf) chunks.push(buf)
  return chunks.join('\r\n')
}

/** ISO date YYYY-MM-DD → YYYYMMDD */
function dateCompact(iso: string): string {
  return iso.replace(/-/g, '')
}

/** Add one day to ISO date YYYY-MM-DD */
function addOneDay(iso: string): string {
  const [y, m, d] = iso.split('-').map(Number)
  const dt = new Date(Date.UTC(y, m - 1, d))
  dt.setUTCDate(dt.getUTCDate() + 1)
  return `${dt.getUTCFullYear()}-${String(dt.getUTCMonth() + 1).padStart(2, '0')}-${String(dt.getUTCDate()).padStart(2, '0')}`
}

/** ISO timestamp (any tz) → YYYYMMDDTHHMMSSZ in UTC */
function timestampUtc(isoTs: string): string {
  const d = new Date(isoTs)
  const pad = (n: number) => String(n).padStart(2, '0')
  return `${d.getUTCFullYear()}${pad(d.getUTCMonth() + 1)}${pad(d.getUTCDate())}T${pad(d.getUTCHours())}${pad(d.getUTCMinutes())}${pad(d.getUTCSeconds())}Z`
}

/** time 'HH:MM:SS' + date 'YYYY-MM-DD' → 'YYYYMMDDTHHMMSS' (no tz suffix — that goes in TZID param) */
function dateTimeLocal(date: string, time: string): string {
  return `${dateCompact(date)}T${time.replace(/:/g, '')}`
}

/** Standard VTIMEZONE-Block für Europe/Zurich (CET/CEST inkl. DST-Regeln) */
export const VTIMEZONE_EUROPE_ZURICH = [
  'BEGIN:VTIMEZONE',
  'TZID:Europe/Zurich',
  'BEGIN:STANDARD',
  'DTSTART:19701025T030000',
  'RRULE:FREQ=YEARLY;BYMONTH=10;BYDAY=-1SU',
  'TZOFFSETFROM:+0200',
  'TZOFFSETTO:+0100',
  'TZNAME:CET',
  'END:STANDARD',
  'BEGIN:DAYLIGHT',
  'DTSTART:19700329T020000',
  'RRULE:FREQ=YEARLY;BYMONTH=3;BYDAY=-1SU',
  'TZOFFSETFROM:+0100',
  'TZOFFSETTO:+0200',
  'TZNAME:CEST',
  'END:DAYLIGHT',
  'END:VTIMEZONE',
].join('\r\n')

/** Build a single VEVENT block */
export function buildVEvent(ev: CalendarEvent, dtstamp: string): string {
  const lines: string[] = ['BEGIN:VEVENT']
  lines.push(`UID:${ev.uid}`)
  lines.push(`DTSTAMP:${dtstamp}`)

  if (ev.timeFrom && ev.timeTo) {
    lines.push(`DTSTART;TZID=Europe/Zurich:${dateTimeLocal(ev.date, ev.timeFrom)}`)
    lines.push(`DTEND;TZID=Europe/Zurich:${dateTimeLocal(ev.date, ev.timeTo)}`)
  } else {
    lines.push(`DTSTART;VALUE=DATE:${dateCompact(ev.date)}`)
    lines.push(`DTEND;VALUE=DATE:${dateCompact(addOneDay(ev.date))}`)
  }

  lines.push(`SUMMARY:${escapeText(ev.summary)}`)
  if (ev.location) lines.push(`LOCATION:${escapeText(ev.location)}`)
  if (ev.description) lines.push(`DESCRIPTION:${escapeText(ev.description)}`)
  lines.push(`STATUS:${ev.status}`)
  lines.push(`LAST-MODIFIED:${timestampUtc(ev.lastModified)}`)
  lines.push('SEQUENCE:0')
  lines.push('END:VEVENT')

  return lines.map(foldLine).join('\r\n')
}

/** Build the full VCALENDAR document */
export function buildVCalendar(opts: {
  calName: string
  events: CalendarEvent[]
}): string {
  const dtstamp = timestampUtc(new Date().toISOString())
  const header = [
    'BEGIN:VCALENDAR',
    'VERSION:2.0',
    'PRODID:-//ATOLL//ical-feed//DE',
    'METHOD:PUBLISH',
    `X-WR-CALNAME:${escapeText(opts.calName)}`,
    'X-WR-TIMEZONE:Europe/Zurich',
    'REFRESH-INTERVAL;VALUE=DURATION:PT1H',
    'X-PUBLISHED-TTL:PT1H',
  ].map(foldLine).join('\r\n')

  const events = opts.events.map((ev) => buildVEvent(ev, dtstamp)).join('\r\n')
  return [header, VTIMEZONE_EUROPE_ZURICH, events, 'END:VCALENDAR'].join('\r\n') + '\r\n'
}
```

```ts
// deno-lint-ignore-file no-explicit-any
import { serve } from 'https://deno.land/std@0.190.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
import { buildVCalendar, type CalendarEvent } from './icalBuilder.ts'

const APP_HOST = Deno.env.get('APP_URL')?.replace(/^https?:\/\//, '') ?? 'dispo.course-director.ch'

serve(async (req) => {
  const url = new URL(req.url)
  const token = url.searchParams.get('token')

  if (!token) {
    return new Response('Missing token query param', { status: 400 })
  }

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!,
  )

  // 1. Token → Instructor
  const { data: instructor, error: instErr } = await supabase
    .from('instructors')
    .select('id, name')
    .eq('calendar_token', token)
    .maybeSingle()

  if (instErr) {
    console.error('[ical-feed] instructor lookup failed:', instErr.message)
    return new Response('Internal error', { status: 500 })
  }
  if (!instructor) {
    return new Response('Unknown token', { status: 404 })
  }

  // 2. Assignments + Courses laden
  const { data: rows, error: qErr } = await supabase
    .from('course_assignments')
    .select(`
      id, role, confirmed, assigned_for_dates, updated_at,
      courses!inner(id, title, status, location, start_date, additional_dates, notes,
                    course_types(code))
    `)
    .eq('instructor_id', instructor.id)
    .gte('courses.start_date', addDays(new Date(), -180))
    .lte('courses.start_date', addDays(new Date(), 730))
    .neq('courses.status', 'cancelled')

  if (qErr) {
    console.error('[ical-feed] assignments query failed for instructor', instructor.id, ':', qErr.message)
    return new Response('Internal error', { status: 500 })
  }

  // 3. course_dates für Zeitslots batch-laden
  const courseIds = Array.from(new Set((rows ?? []).map((r: any) => r.courses?.id).filter(Boolean)))
  const courseDatesByKey = new Map<string, { from: string | null; to: string | null }>()
  if (courseIds.length > 0) {
    const { data: cdRows, error: cdErr } = await supabase
      .from('course_dates')
      .select('course_id, date, time_from, time_to')
      .in('course_id', courseIds)
    if (cdErr) {
      console.error('[ical-feed] course_dates query failed:', cdErr.message)
    } else {
      for (const cd of cdRows ?? []) {
        courseDatesByKey.set(`${cd.course_id}|${cd.date}`, {
          from: cd.time_from,
          to: cd.time_to,
        })
      }
    }
  }

  // 4. Pro (assignment, date) ein Event bauen
  const events: CalendarEvent[] = []
  for (const a of rows ?? []) {
    const c = a.courses
    if (!c) continue

    // Daten ermitteln: explicit assigned_for_dates ODER alle Course-Daten
    let dates: string[]
    if (Array.isArray(a.assigned_for_dates) && a.assigned_for_dates.length > 0) {
      dates = a.assigned_for_dates as string[]
    } else {
      dates = [c.start_date, ...((c.additional_dates as string[] | null) ?? [])]
    }

    for (const date of dates) {
      const slot = courseDatesByKey.get(`${c.id}|${date}`)
      const courseTypeCode = c.course_types?.code ?? null
      const descParts = [
        `Status: ${c.status}`,
        courseTypeCode ? `Kurstyp: ${courseTypeCode}` : null,
        c.notes ? `Notizen: ${c.notes}` : null,
      ].filter(Boolean)

      events.push({
        uid: `assignment-${a.id}-${date.replace(/-/g, '')}@${APP_HOST}`,
        summary: `${c.title} (${a.role})`,
        location: c.location,
        description: descParts.join('\n'),
        date,
        timeFrom: slot?.from ?? null,
        timeTo: slot?.to ?? null,
        status: c.status === 'tentative' ? 'TENTATIVE' : 'CONFIRMED',
        lastModified: a.updated_at,
      })
    }
  }

  // 5. .ics zusammenbauen
  const ics = buildVCalendar({
    calName: `ATOLL — ${instructor.name}`,
    events,
  })

  return new Response(ics, {
    status: 200,
    headers: {
      'Content-Type': 'text/calendar; charset=utf-8',
      'Cache-Control': 'no-cache, must-revalidate',
    },
  })
})

/** YYYY-MM-DD String, Δ Tage */
function addDays(d: Date, delta: number): string {
  const x = new Date(d)
  x.setUTCDate(x.getUTCDate() + delta)
  const pad = (n: number) => String(n).padStart(2, '0')
  return `${x.getUTCFullYear()}-${pad(x.getUTCMonth() + 1)}-${pad(x.getUTCDate())}`
}
```

```ts
// ============================================================
// Calendar-Token (für iCal-Feed-Subscription im MyProfileScreen)
// ============================================================

export async function fetchMyCalendarToken(instructorId: string): Promise<string | null> {
  const { data, error } = await supabase
    .from('instructors')
    .select('calendar_token')
    .eq('id', instructorId)
    .maybeSingle()
  if (error) throw error
  return (data?.calendar_token as string | null) ?? null
}

export async function rotateMyCalendarToken(): Promise<string> {
  const { data, error } = await supabase.rpc('rotate_calendar_token')
  if (error) throw error
  if (typeof data !== 'string') throw new Error('rotate_calendar_token: unexpected response')
  return data
}

/**
 * Baut den iCal-Feed-URL für einen Token.
 * Nimmt SUPABASE_URL aus dem Client-Env (via import.meta.env).
 */
export function buildCalendarFeedUrl(token: string): string {
  const supabaseUrl = (import.meta as any).env.VITE_SUPABASE_URL as string
  return `${supabaseUrl}/functions/v1/ical-feed?token=${encodeURIComponent(token)}`
}
```

```tsx
/**
 * CalendarSync — Sektion im MyProfileScreen für iCal-Feed-Subscription.
 * Zeigt den Feed-URL mit Copy-Button + Rotate-Funktion + Apple-Calendar-Anleitung.
 */

import { useEffect, useState } from 'react'
import { useTranslation } from 'react-i18next'
import { Icon } from '@/foundation'
import {
  fetchMyCalendarToken,
  rotateMyCalendarToken,
  buildCalendarFeedUrl,
} from '@/lib/queries'

interface Props {
  instructorId: string
}

export function CalendarSync({ instructorId }: Props) {
  const { t } = useTranslation()
  const [token, setToken] = useState<string | null>(null)
  const [loading, setLoading] = useState(true)
  const [copied, setCopied] = useState(false)
  const [helpOpen, setHelpOpen] = useState(false)

  useEffect(() => {
    let cancelled = false
    setLoading(true)
    fetchMyCalendarToken(instructorId)
      .then((t) => { if (!cancelled) setToken(t) })
      .catch((err) => console.error('[calendar-sync] load token failed', err))
      .finally(() => { if (!cancelled) setLoading(false) })
    return () => { cancelled = true }
  }, [instructorId])

  const url = token ? buildCalendarFeedUrl(token) : ''

  async function copyUrl() {
    if (!url) return
    try {
      await navigator.clipboard.writeText(url)
      setCopied(true)
      setTimeout(() => setCopied(false), 2000)
    } catch (err) {
      console.error('[calendar-sync] clipboard write failed', err)
    }
  }

  async function rotate() {
    if (!confirm(t('my_profile.calendar_rotate_confirm'))) return
    try {
      const newToken = await rotateMyCalendarToken()
      setToken(newToken)
    } catch (err) {
      console.error('[calendar-sync] rotate failed', err)
      alert(String(err))
    }
  }

  return (
    <section className="atoll-cockpit__card">
      <h2 className="atoll-cockpit__card-title">{t('my_profile.calendar_section_title')}</h2>
      <p className="atoll-cockpit__card-sub">{t('my_profile.calendar_subtitle')}</p>

      {loading ? (
        <div className="caption-2" style={{ marginTop: 8 }}>…</div>
      ) : (
        <>
          <div style={{ display: 'flex', gap: 8, marginTop: 12, alignItems: 'center' }}>
            <input
              type="text"
              value={url}
              readOnly
              onFocus={(e) => e.currentTarget.select()}
              style={{
                flex: 1,
                padding: '8px 10px',
                borderRadius: 8,
                border: '1px solid var(--border-tertiary)',
                background: 'var(--bg-card)',
                color: 'var(--text-primary)',
                font: 'inherit',
                fontSize: 12,
                fontFamily: 'monospace',
              }}
            />
            <button
              type="button"
              className="atoll-btn"
              onClick={copyUrl}
              title={t('my_profile.calendar_copy_button')}
            >
              <Icon.Copy size={14} />
              {copied ? ' ✓' : ` ${t('my_profile.calendar_copy_button')}`}
            </button>
          </div>

          <div style={{ marginTop: 12 }}>
            <button
              type="button"
              className="atoll-btn"
              onClick={() => setHelpOpen((v) => !v)}
              style={{ fontSize: 12 }}
            >
              {helpOpen ? '▾ ' : '▸ '}{t('my_profile.calendar_apple_help_title')}
            </button>
            {helpOpen && (
              <div className="caption-2" style={{ marginTop: 6, lineHeight: 1.5 }}>
                <div>{t('my_profile.calendar_apple_help_macos')}</div>
                <div>{t('my_profile.calendar_apple_help_ios')}</div>
              </div>
            )}
          </div>

          <div style={{ marginTop: 12 }}>
            <button
              type="button"
              className="atoll-btn"
              onClick={rotate}
              style={{ fontSize: 12, color: 'var(--text-danger, #c00)' }}
            >
              {t('my_profile.calendar_rotate_button')}
            </button>
          </div>
        </>
      )}
    </section>
  )
}
```
